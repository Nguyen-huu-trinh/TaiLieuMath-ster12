            \foreach \i in {1,...,#1} {
                \draw[softgray, line width=0.2pt] (0,-{\i}) -- (\textwidth-4pt,-{\i});

        \end{tikzpicture}

    \fi
    % Đã xóa hoàn toàn khoảng cách thừa ở nhánh else
